9   public class  Lager
10  {
11      private Artikel artikelTab [];
12      private int anzArtikel;
13  
14     public Lager (int maxAnzArtikel) {
15          artikelTab = new Artikel[maxAnzArtikel];
16          for (int i = 0 ; i < maxAnzArtikel ; i++){
17              artikelTab[i] = null;
18          }
19          anzArtikel = 0;
20     }
21      public Lager () {
22          artikelTab = new Artikel[10];
23          for (int i = 0 ; i < 10 ; i++){
24              artikelTab[i] = null;
25          }
26          anzArtikel = 0;
27     }
28  
29      /**
30       * Lager_Objekt als Zeichenkette aufbereiten
31       *
32       * @return Zeichenkette
33       */
34     private  int get_index_artikel (int artikelNr){
35         for (int i = 0 ; i < anzArtikel ; i++){
36             if(artikelTab[i].getArtikelNr() == artikelNr){
37                 return i;
38             }
39         }
40         return -1;
41     }
42      /**
43       * Methode zum legen neuen Artikel
44       *
45       *
46       *
47       */
48     public void legeAnArtikel(Artikel artikel){
49         if(anzArtikel >= artikelTab.length || artikel == null){
50             throw new IllegalAccessError("Der lager ist Voll");
51         }
52         artikelTab[anzArtikel] = artikel;
53         anzArtikel++;
54     }
55      /**
56       * Entferne eines Artikel aus dem Lager
57       *
58       * @param artikelNr zu uebergebende Artikelnummer
59       */
60     public void entferneArtikel(int artikelNr){
61         int i = get_index_artikel(artikelNr);
62      if(i > anzArtikel   || i >= artikelTab.length || i == -1){
63          throw new IllegalAccessError("Es ist nicht verfuegbar");
64      }
65      for ( ; i < anzArtikel ; i++){
66          artikelTab[i] = artikelTab[i+1];
67      }
68      artikelTab[anzArtikel--] = null;
69     }

71      /**
72       * Zugang buchen f√ºr einen Artikel
73       *
74       * @param artikelNr   zu uebergebende Artikelnummer
75       * @param menge       zu buchende Menge > 0
76       */
77     public void bucheZugang(int artikelNr, int menge){
78      int artikel_index = get_index_artikel(artikelNr);
79      if (artikel_index == -1) {
80          throw new IllegalAccessError("Es ist nicht verfuegbar");
81      }
82      artikelTab[artikel_index].bucheZugang(menge);
83     }

85      /**

88       * @param artikelNr   zu uebergebende Artikelnummer
89       * @param menge       zu buchende Menge > 0
90       */
91     public void bucheAbgang(int artikelNr, int menge){
92         int artikel_index = get_index_artikel(artikelNr);
93         if (artikel_index == -1) {
94             throw new IllegalAccessError("Es ist nicht verfuegbar");
95         }
96         artikelTab[artikel_index].bucheAbgang(menge);
97     }

99      /**
100      * Methode zum Ermitteln eines Artikels an einer bestimmten Stelle im Lager
101      *
102      *
103      * @return  index
104      */
105    public Artikel getArtikel(int index){
106       if (index < 0 || index >= anzArtikel){
107           throw new IllegalAccessError("Diese Index ist nicht verfuegbar");
108       }
109        return artikelTab[index];
110    }

112     /**
113      *Preis f√ºr alle Artikel um einen bestimmten Protzentsatz ver√§ndern
114      *
115      * @param prozent
116      */
117    public void  aenderePreisAllerArtikel(double prozent){
118 
119        for (int i=0 ; i <anzArtikel ; i++){
120            double prise = artikelTab[i].getPreis();
121            double neu_preis = prise + prise * prozent ;
122            artikelTab[i].setPreis(neu_preis);
123 
124        }
125    }
126    public String toString(){
127        StringBuilder temp = new StringBuilder("Grosse der Lager: " + artikelTab.length +
128                " Anzahl der Artikel:" + anzArtikel);
129        for (int i = 0 ; i < anzArtikel ; i++){
130            temp.append("\n").append(i).append(".\t");
131            temp.append(artikelTab[i].toString());
132        }
133        return temp.toString();
134    }
